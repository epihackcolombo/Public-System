439  //Update UIs based on server response
440  public void updateViews(String result, int action) {
441 
442  try {
443  if (GlobalMethods.validateString(result)) {
444 
445  //validation block ------------------------
446  //error_net_connection - no route to host, can't find an active connection
447  //error_net_other - network time-out or other connector exceptions
448  //error_db_params - inappropriate parameters with the request
449  //error_db_connect - database connection issue at server side
450  //authentication_required - user need to be authenticate
451  //authentication_expired - user session has expired
452 
453  if (result.contains("{'status':")) { //In order to support backward-compatibility
454 
455  Gson gson = new Gson();
456  ProfileLoginPojo plp = gson.fromJson(result, ProfileLoginPojo.class);
457 
458  if (GlobalMethods.validateString(plp.getStatus())) {
459 
460  if (("error_net_connection").equals(plp.getStatus().toLowerCase(Locale.ENGLISH))) {
461 
462  popdRequestFailed(getString(R.string.msg_connect_no_title1), getString(R.string.msg_connect_no_para1), action);
463  }
464  else if (("error_net_other").equals(plp.getStatus().toLowerCase(Locale.ENGLISH))) {
465 
466  popdRequestFailed(getString(R.string.msg_connect_no_title1), getString(R.string.msg_connect_no_para2), action);
467  }
468  else if (("error_db_params").equals(plp.getStatus().toLowerCase(Locale.ENGLISH))) {
469 
470  popdRequestFailed(getString(R.string.msg_response_unexpected_title2), getString(R.string.msg_response_unexpected_para2), action);
471  } else if (("error_db_connect").equals(plp.getStatus().toLowerCase(Locale.ENGLISH))) {
472 
473  popdRequestFailed(getString(R.string.msg_response_unexpected_title2), getString(R.string.msg_response_unexpected_para2), action);
474  } else if (("authentication_required").equals(plp.getStatus().toLowerCase(Locale.ENGLISH))) {
475 
476  popdReauth(getString(R.string.msg_response_reauthentication_title1), getString(R.string.msg_response_reauthentication_para1), action);
477  } else if (("authentication_expired").equals(plp.getStatus().toLowerCase(Locale.ENGLISH))) {
478
